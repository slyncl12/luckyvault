async withdrawFromNavi(userAddress: string, amount: number): Promise<any> {
  const { withdrawCoinPTB } = await import('@naviprotocol/lending');
  const { Transaction } = await import('@mysten/sui/transactions');
  
  const USDC_TYPE = '0xdba34672e30cb065b1f93e3ab55318768fd6fef66c15942c9f7cb846e2f900e7::usdc::USDC';
  
  // Create transaction
  const tx = new Transaction();
  
  // Withdraw from NAVI (SDK handles AccountCap automatically)
  await withdrawCoinPTB(
    tx,
    USDC_TYPE,
    Math.floor(amount * 1_000_000)
  );
  
  // Transaction is ready to be executed by caller
  return tx;
}
